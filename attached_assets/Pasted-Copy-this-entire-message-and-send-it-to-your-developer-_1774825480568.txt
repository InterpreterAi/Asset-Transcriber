3. When a final segment arrives

Commit the preview into the transcript and reset preview.

function onFinalSegment(text, speaker) {
  setSegments(prev => {
    const last = prev[prev.length - 1]

    // merge if same speaker
    if (last && last.speaker === speaker) {
      const updated = [...prev]
      updated[updated.length - 1] = {
        ...last,
        text: last.text + " " + text
      }
      return updated
    }

    return [...prev, { speaker, text }]
  })

  setActivePreviewLine("")
}
